动画(animation)：

--------------------------------

@keyframes myfirst
{
	from {left:0px;} /* 0% */
	to {left:200px;} /* 100% */
}
@-moz-keyframes myfirst /* Firefox */
{
	from {left:0px;}
	to {left:200px;}
}
@-webkit-keyframes myfirst /* Safari 和 Chrome */
{
	from {left:0px;}
	to {left:200px;}
}
@-o-keyframes myfirst /* Opera */
{
	from {left:0px;}
	to {left:200px;}
}

.test{
	-webkit-animation: myfirst 4s linear 1s infinite; 
	        animation: myfirst 4s linear 1s infinite;
}
